Note

Untuk pemasangan script bot whatsapp via termux butuh penyimpanan yang besar dan kuota yang cukup. Script bot WhatsApp dibagi jadi 2 yaitu case dan plugin

Contoh kode plugin

let handler = async (m, { conn, text }) => {
    let [l, r] = text.split`|`
    if (!l) l = ''
    if (!r) r = ''
    conn.reply(m.chat, l + readMore + r, m)
}
handler.menu = ['readmore', 'spoiler'].map(v => v + ' *|*')
handler.tags = ['tools']
handler.command = /^(spoiler|hidetext|readmore|selengkapnya)$/i

export default handler

const more = String.fromCharCode(8206)
const readMore = more.repeat(4001);
